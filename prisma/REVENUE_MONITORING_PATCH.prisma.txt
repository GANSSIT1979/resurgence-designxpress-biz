// Add this model to your Prisma schema.

model RevenueRecord {
  id                 String   @id @default(cuid())
  saleDate           DateTime @default(now())
  sourceType         String
  title              String
  referenceNo        String?
  creatorProfileId   String?
  partnerId          String?
  creatorName        String?
  partnerName        String?
  sponsorName        String?
  productCategory    String?
  merchandiseType    String?
  quantity           Int?
  unitPrice          Float?
  grossSales         Float    @default(0)
  costAmount         Float?   @default(0)
  netRevenue         Float?   @default(0)
  sponsorshipPercent Float?   @default(0)
  talentFeeAmount    Float?   @default(0)
  franchiseAmount    Float?   @default(0)
  companyShareAmount Float?   @default(0)
  notes              String?
  attachmentUrl      String?
  createdAt          DateTime @default(now())
  updatedAt          DateTime @updatedAt

  @@index([saleDate])
  @@index([sourceType])
  @@index([creatorProfileId])
  @@index([partnerId])
  @@index([creatorName])
  @@index([partnerName])
}
